97 private void Calculate(AStarNode fromNode, AStarNode toNode, TransitionType transitionType = TransitionType.Path)
98 {
99 _closedList.Add(fromNode);
100 foreach (var connected in fromNode.Node.ConnectedNode)
101 {
102 AStarNode connectedAStarNode = new AStarNode(connected, fromNode.Node);
103 if (_closedList.Contains(connectedAStarNode))
104 {
105 continue;
106 }
107
108 // Link로 인접한 두 노드 모두 동일한 TransitionType에, Path가 아닌 경우
109 if (fromNode.Node.TransitionType == connectedAStarNode.Node.TransitionType && fromNode.Node.TransitionType != TransitionType.Path)
110 {
111 // 지정된 TransitionType이 아니라면, 안내하지 않음
112 if (fromNode.Node.TransitionType != transitionType)
113 {
114 // 이 곳으로는 절대로 안내하면 안되니깐 오픈리스트에서 제외
115 _openList.Remove(connectedAStarNode);
116 continue;
117 }
118 }
119
120 // Calculate weight
121 connectedAStarNode.G = fromNode.G + connectedAStarNode.Connected.Distance;
122 connectedAStarNode.H = (toNode.Node.Position - connectedAStarNode.Node.Position).magnitude;
123
124 if (_openList.Contains(connectedAStarNode))
125 {
126 var openedAStarNode = _openList.Single(target => target.Node.ID == connectedAStarNode.Node.ID);
127
128 if (openedAStarNode != null)
129 {
130 // If the calculated weight is greater than the existing weight, skip
131 if (openedAStarNode.F < connectedAStarNode.F)
132 {
133 continue;
134 }
135 }
136 _openList.Remove(connectedAStarNode);
137 }
138 _openList.Add(connectedAStarNode);
139 }

149 internal static Path FindPath(INode startNode, INode endNode, TransitionType transitionType)
150 {
151 AStarNode from = new AStarNode(startNode);

153
154 AStarNode fromNode = from;
155 AStarNode toNode = to;
156
157 AStarPathGenerator generator = new AStarPathGenerator();
158 do
159 {
160 // Path calculation
161 generator.Calculate(fromNode, toNode, transitionType);
162
163 fromNode = generator._openList.OrderBy(node => node.F).FirstOrDefault();
164 if (fromNode == null)
165 {
166 if (generator._openList.Contains(toNode))
167 {
168 break;
169 }
170
171 return null;
172 }
173 generator._openList.Remove(fromNode);
174 }
175 while (!generator._closedList.Contains(toNode));
176
177 AStarNode currentNode = generator._closedList.Last();
178 List<AStarNode> retNodes = new List<AStarNode>();
179
180 while (currentNode != from)
181 {
182 retNodes.Add(currentNode);
183 currentNode = generator._closedList.FirstOrDefault(node => node.Node.ID == currentNode.FromNode.ID);
184 }
185 retNodes.Add(from);
186 retNodes.Reverse();
187
188 Path path = new Path();
189 foreach (var found in retNodes)
190 {
191 if (found.Node != null && found.FromNode != null)
192 {
193 path.Add(found.Connected);
194 }
195 path.Add(found.Node);
196 }
197
198 return path;
