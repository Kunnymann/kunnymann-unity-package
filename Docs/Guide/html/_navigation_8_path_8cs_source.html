216 private float GetDistanceFromLink(Vector3 point1_InLink, Vector3 point2_InLink, Vector3 target, out Vector3 projectPoint)
217 {
218 // Target으로부터 Link에 투영된 Point
219 projectPoint = FindProjectedPointInLine(point1_InLink, point2_InLink, target);
220
221 // 해당 Point가 Link 위에 존재하는지 확인
222 if (projectPoint.x >= Mathf.Min(point1_InLink.x, point2_InLink.x) && projectPoint.x <= Mathf.Max(point1_InLink.x, point2_InLink.x) &&
223 projectPoint.z >= Mathf.Min(point1_InLink.z, point2_InLink.z) && projectPoint.z <= Mathf.Max(point1_InLink.z, point2_InLink.z))
224 {
225 // Link와의 거리 계산
226 return Vector3Utility.GetXZDistance(target, projectPoint);
227 }

231 float weight1 = Vector3Utility.GetXZDistance(target, point1_InLink);
232 float weight2 = Vector3Utility.GetXZDistance(target, point2_InLink);
233 if (weight1 < weight2)

236 }

246 private Vector3 FindProjectedPointInLine(Vector3 pointInLine1, Vector3 pointInLine2, Vector3 target)
247 {
248 Vector3 p = target;
249
250 float m1, k1, m2, k2;
251
252 if (pointInLine1.x == pointInLine2.x) // 선분이 x축에 평행한 case
253 {
254 p.x = pointInLine1.x;
255 p.z = target.z;
256 }
257 else if (pointInLine1.z == pointInLine2.z) // 선분이 z축에 평행한 case
258 {
259 p.x = target.x;
260 p.z = pointInLine1.z;
261 }
262 else // 일반적인 case
263 {
264 m1 = (pointInLine1.z - pointInLine2.z) / (pointInLine1.x - pointInLine2.x);
265 k1 = pointInLine1.z - m1 * pointInLine1.x;
266
267 m2 = -1 / m1;
268 k2 = target.z - m2 * target.x;
269
270 p.x = (k2 - k1) / (m1 - m2);
271 p.z = m1 * p.x + k1;
272 }
273
274 return p;
